CLI Management
lxc launch ubuntu:16.04 my-ubuntu
lxc exec my-ubuntu -- /bin/bash
lxc exec my-ubuntu -- apt-get update
lxc file pull my-ubuntu/etc/hosts .
lxc file push hosts my-ubuntu/tmp/
lxc stop my-ubuntu
lxc delete my-ubuntu
