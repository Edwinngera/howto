mws = MWSConnection()
feedtype = ‘_POST_ORDER_ACKNOWLEDGEMENT_DATA_’
content = ‘<xml>your feed data</xml>'
response = mws.submit_feed(FeedType=feedtype, FeedContent=content,
                                              content_type='text/xml’)
info = response.SubmitFeedResult.FeedSubmissionInfo
print info.FeedType, info.FeedProcessingStatus, info.FeedSubmissionId

# wait 10mins for Amazon to process the feed...
time.sleep(600)

report = mws.get_feed_submission_result(FeedSubmissionId=info.FeedSubmissionId)
print report
