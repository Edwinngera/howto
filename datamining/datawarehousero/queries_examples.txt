
SELECT count(*) as Total, AutoritateContractanta as 'Autoritate Contractanta', TipContract, TipProcedura FROM `contracte` GROUP BY AutoritateContractanta, TipContract, TipProcedura ORDER BY total DESC LIMIT 40

SELECT SUM(ValoareEUR) FROM `contracte` WHERE `AutoritateContractanta` LIKE '%ROMSILVA%' AND `DataAnuntAtribuire` LIKE '%2015%' ORDER BY `AutoritateContractanta` ASC
SELECT Count(*), YEAR( DATE_ADD( DataAnuntAtribuire, INTERVAL 7200 SECOND )) FROM `contracte` WHERE `AutoritateContractanta` LIKE '%ROMSILVA%' GROUP BY YEAR( DATE_ADD( DataAnuntAtribuire, INTERVAL 7200 SECOND ))
SELECT COUNT(*) AS TOTAL, ROUND(SUM(ValoareEUR),2) AS SUMA, YEAR( DATE_ADD( DataAnuntAtribuire, INTERVAL 7200 SECOND )) AS ANUL FROM `contracte` WHERE `AutoritateContractanta` LIKE '%ROMSILVA%' GROUP BY YEAR( DATE_ADD( DataAnuntAtribuire, INTERVAL 7200 SECOND ))
